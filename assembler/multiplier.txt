			lsr r4, r4, 7;

			and r0, r4, r12;
			lsr r12, r12, 5;
			and r1, r4, r12;

multiply:	str lr, sp, 0;
			addi sp, sp, 1;
			addi r2, r_zero, 0;
			addi r3, r_zero, 0;
			beq r0, r_zero, exitMult;
multLoop:	beq r1, r3, exitMult;
			add r2, r2, r0;
			addi r3, r3, 1;
			b multLoop;

			addi sp, sp, -1;
			ldr lr, sp, 0;
			br lr;
